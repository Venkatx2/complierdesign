%{
#include<stdio.h>
int count = 0;
%}

%%
[0-9]+              { count++; }        /* Numbers */
[a-zA-Z_][a-zA-Z0-9_]*   { count++; }  /* Identifiers */
[+\-*/=]            { count++; }        /* Operators */
[ \t\n]             ;                   /* Ignore spaces */
.                   { count++; }        /* Other symbols */
%%

int yywrap()
{
    return 1;
}

int main()
{
    printf("Enter the input string:\n");
    yylex();
    printf("\nTotal number of tokens = %d\n", count);
    return 0;
}
